  <li><DocsLink href="/hooks/use-throttle">useThrottle</DocsLink> — Returns a throttled copy of a changing value.</li>
  <li><DocsLink href="/hooks/use-throttle-effect">useThrottleEffect</DocsLink> — Runs an effect at throttled intervals for dependency changes.</li>
  <li><DocsLink href="/hooks/use-throttle-fn">useThrottleFn</DocsLink> — Returns a throttled function wrapper.</li>
  <li><DocsLink href="/hooks/use-tracked-effect">useTrackedEffect</DocsLink> — Shows which dependency indexes changed between runs.</li>

  <li><DocsLink href="/hooks/use-why-did-you-update">useWhyDidYouUpdate</DocsLink> — Logs prop differences to the console for debugging.</li>
  <li><DocsLink href="/hooks/use-isomorphic-layout-effect">useIsomorphicLayoutEffect</DocsLink> — useLayoutEffect in the browser and useEffect on the server.</li>
  <li><DocsLink href="/hooks/use-mount">useMount</DocsLink> — Runs logic once when the component mounts.</li>
  <li><DocsLink href="/hooks/use-unmount">useUnmount</DocsLink> — Runs cleanup logic when the component unmounts.</li>
  <li><DocsLink href="/hooks/use-unmounted-ref">useUnmountedRef</DocsLink> — Mutable ref that flips to true after unmount.</li>
  <li><DocsLink href="/hooks/use-click-away">useClickAway</DocsLink> — Calls a handler when interaction happens outside the target.</li>
  <li><DocsLink href="/hooks/use-drag">useDrag</DocsLink> — Makes an element draggable and writes payload data to dataTransfer.</li>

  <li><DocsLink href="/hooks/use-event-listener">useEventListener</DocsLink> — Attaches DOM event listeners with cleanup.</li>
  <li><DocsLink href="/hooks/use-focus-within">useFocusWithin</DocsLink> — Reports whether focus is currently inside a subtree.</li>
  <li><DocsLink href="/hooks/use-fullscreen">useFullscreen</DocsLink> — Controls fullscreen mode for a specific target element.</li>
  <li><DocsLink href="/hooks/use-hover">useHover</DocsLink> — Tracks whether a target is hovered.</li>
  <li><DocsLink href="/hooks/use-key-press">useKeyPress</DocsLink> — Runs a handler when matching keys are pressed.</li>
  <li><DocsLink href="/hooks/use-long-press">useLongPress</DocsLink> — Recognizes long-press gestures with delay control.</li>
  <li><DocsLink href="/hooks/use-mouse">useMouse</DocsLink> — Tracks pointer coordinates relative to the page and an element.</li>
  <li><DocsLink href="/hooks/use-mutation-observer">useMutationObserver</DocsLink> — Subscribes to DOM mutations on a target.</li>
  <li><DocsLink href="/hooks/use-scroll">useScroll</DocsLink> — Reads scroll position from an element or document.</li>
  <li><DocsLink href="/hooks/use-size">useSize</DocsLink> — Reads width and height from ResizeObserver.</li>
  <li><DocsLink href="/hooks/use-text-selection">useTextSelection</DocsLink> — Tracks current text selection and bounding box.</li>
  <li><DocsLink href="/hooks/use-cookie-state">useCookieState</DocsLink> — Persists a string state value in a browser cookie.</li>
  <li><DocsLink href="/hooks/use-document-visibility">useDocumentVisibility</DocsLink> — Tracks document.visibilityState.</li>
  <li><DocsLink href="/hooks/use-external">useExternal</DocsLink> — Loads external JS or CSS and reports the loading status.</li>
  <li><DocsLink href="/hooks/use-favicon">useFavicon</DocsLink> — Sets the current document favicon.</li>
  <li><DocsLink href="/hooks/use-in-viewport">useInViewport</DocsLink> — Tracks visibility with IntersectionObserver.</li>
